# Divider
A divider is a line used to group content on lists and layouts.

## Horizontal line
Set the wrapper Container to display:flex

<Playground>
  <ThemeContainer>
    <Parent>
      <Container>
        <Ulist>
          <li>item 1 item 1</li>
          <li>item 2</li>
          <Divider
            component="li"
            margin="20px"
            thickness="1px"
          />
          <li>item 3</li>
          <li>item 4</li>
          <li>item 5</li>
        </Ulist>
      </Container>
    </Parent>
  </ThemeContainer>
</Playground>

## Inside a modal

<Playground>
{
  () => {
    const [visible, setVisibility] = React.useState(false)
    const onCancel = React.useCallback(() => {
      setVisibility(false);
    }, [])
    const onOk = React.useCallback(() => {
      setVisibility(false);
    }, [])
    return (
      <ThemeContainer>
        <Parent>
          <ContainerModal>
            <Button onClick={() => setVisibility(true)}>Open Modal</Button>
            <Modal
              title={'Title'}
              onCancel={onCancel}
              onOk={onOk}
              visible={visible}
            >
              <ModalContent>
                <div>Some content here</div>
                <Divider
                  margin="20px"
                  thickness="1px"
                />
                <div>More content here</div>
                <Divider
                  margin="20px"
                  thickness="1px"
                />
                <div>More content here</div>
              </ModalContent>
            </Modal>
          </ContainerModal>
        </Parent>
      </ThemeContainer>)
  }
}
</Playground>

## With text
