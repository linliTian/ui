# Floater
Floaters all users to display information with respect to an anchor point

## Simple

Add a simple floater to any anchor point using a `ref`.

<Playground>
{
  () => {
    const [buttonRef, setRef] = React.useState(null);
    const [open, setOpen] = React.useState(false);
    const handleClick = React.useCallback(() => {
      if (open) {
        setOpen(false);
      } else {
        setOpen(true);
      }
    }, [setOpen, open]);
    const handleRef = React.useCallback(
      (node) => {
        setRef(node);
      },
      [setRef]
    );
    return (
      <ThemeContainer>
        <React.Fragment>
          <Button onClick={handleClick} ref={handleRef}>
            {open ? 'Hide' : 'Show'}
          </Button>
          <Floater
            anchorElement={buttonRef}
            position={['bc', 'tc']}
            open={open && buttonRef !== null}
          >
            I am the content
          </Floater>
        </React.Fragment>
      </ThemeContainer>
    )
  }
}
</Playground>

## Complex

A floater can "float" around the screen, simply by change the anchor point. The target anchor can be retrieved from the `click` event on the button.

<Playground>
  {
    () => {
      const [hasMountedFloater, setHasMountedFloater] = React.useState(false)
      const [anchorElement, setAnchorElement] = React.useState(null)
      const [floaterPosition, setFloaterPosition] = React.useState([
        'bc',
        'tc',
      ])
      const handleButtonClick = (e, position) => {
        const element = e.target;
        if (element === anchorElement) {
          setAnchorElement(null);
          setHasMountedFloater(false);
        } else if (anchorElement) {
          setAnchorElement(element);
          setFloaterPosition(position);
          setHasMountedFloater(true);
        } else {
          setAnchorElement(element);
          setFloaterPosition(position);
        }
      }
      return (
        <ThemeContainer>
          <ButtonContainer>
              <Button onClick={(e) => handleButtonClick(e, ['tc', 'bc'])}>
                Move to Top
              </Button>
              <span style={{ display: 'inline-block', width: '20px' }} />
              <Button onClick={(e) => handleButtonClick(e, ['bc', 'tc'])}>
                Move to Bottom
              </Button>
              <span style={{ display: 'inline-block', width: '20px' }} />
              <Button onClick={(e) => handleButtonClick(e, ['cr', 'cl'])}>
                Move to Right
              </Button>
            </ButtonContainer>
            <Floater
              position={floaterPosition}
              anchorElement={anchorElement}
              open={anchorElement !== null}
              animationProps={{
                initial: { opacity: 0 },
                animate: { opacity: 1 },
                exit: { opacity: 0 },
                layout: hasMountedFloater,
              }}
              matchAnchorWidth
            >
              <Container />
            </Floater>
        </ThemeContainer>
      )
  }
}
</Playground>
